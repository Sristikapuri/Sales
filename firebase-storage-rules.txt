// Firebase Storage Rules for Beauty Hub
// Copy these rules to Firebase Console > Storage > Rules

rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    // Allow read/write access to product images for authenticated users (including anonymous)
    match /product_images/{imageId} {
      allow read, write: if request.auth != null;
    }
    
    // Allow read/write access to test files for diagnostics
    match /test/{testFile} {
      allow read, write: if request.auth != null;
    }
    
    // For development - allow all access (remove in production)
    // match /{allPaths=**} {
    //   allow read, write: if true;
    // }
  }
}
